# GDAL (Geospatial Data Abstraction Library)
# Note importante: GDAL est souvent mieux installé via le gestionnaire de paquets système
# plutôt que via pip pour éviter les problèmes de dépendances binaires.
#
# Options d'installation recommandées:
# - Ubuntu/Debian: sudo apt-get install python3-gdal gdal-bin
# - macOS: brew install gdal
# - Windows: conda install -c conda-forge gdal
# - Alternative: pip install gdal (peut nécessiter des compilateurs)
#
# Version minimale: 3.0.0 (pour support OpenFileGDB et fonctionnalités modernes)
gdal>=3.0.0

# Barres de progression pour l'affichage du statut de conversion
# Utilisé pour afficher la progression globale et par couche
